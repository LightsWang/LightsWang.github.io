# 坑点

# 1、在app.js

中可以处理一些登陆，以及全局变量数据，需要注意的是有可能在app.js中的请求比子页面中的请求还慢，一般可以加启动页，在启动页之后跳转子页面，如果没有启动页，必须在app.js中写这些请求逻辑，也有方法处理，可以使用回调函数，保证子页面请求在app.js请求之后

  App({
  onLaunch: function () {
    this.login();
    wx.getSystemInfo({
      success: e => {
        this.globalData.StatusBar = e.statusBarHeight;
        let capsule = wx.getMenuButtonBoundingClientRect();
        if (capsule) {
          this.globalData.Custom = capsule;
          this.globalData.CustomBar = capsule.bottom + capsule.top - e.statusBarHeight;
        } else {
          this.globalData.CustomBar = e.statusBarHeight + 50;
        }
      }
    })
  },
  //全局数据
  globalData: {
    host: 'https://XXXXXXXXXXXXXXXXXX',
    checkLogin: false, // 登录是否请求完毕
    openid: '',
    unionid: '',
    visit_face_url: '', // 人脸图片url
    guestApplyMsg: {} // 访客申请信息
  },
  login() {
    let that = this;
    wx.login({
      success: function (res) {
        if (res.code) {
          let params = {
            user_code: res.code
          }
          wx.getSystemInfo({
            success: function (info) {
              console.log(info);
              if (info.environment && info.environment === 'wxwork') { // 企业微信
                wx.qy.login({
                  success: function (qy) {
                    if (qy.code) {
                      params.qywx_code = qy.code;
                      that.loginRequest(params)
                    } else {
                      wx.showToast({
                        title: '获取用户登录态失败！',
                        icon: 'none',
                        duration: 2000
                      })
                    }
                  }
                })
              } else { // 微信
                that.loginRequest(params)
              }
            }
          })
        } else {
          wx.showToast({
            title: '获取用户登录态失败！',
            icon: 'none',
            duration: 2000
          })
        }

      }
    })
  },
  // 请求后端
  loginRequest(params) {
    let that = this;
    wx.showLoading({
      title: '加载中',
      mask: true
    })
    wx.request({
      url: that.globalData.host + '/api/v1/getwxopenid',
      method: 'POST',
      data: params,
      success: function (res) {
        if (res.data && res.data.errcode === 0) {

          //存入缓存中备用
          that.globalData.openid = res.data.openid;
          that.globalData.unionid = res.data.unionid;

          // 由于这里是网络请求，可能会在 Page.onLoad 之后才返回
          // 所以此处加入 callback 以防止这种情况
          that.globalData.checkLogin = true;
          if (that.checkLoginReadyCallback) {
            that.checkLoginReadyCallback();
          }

        } else {
          wx.showToast({
            title: '获取用户登录态失败！',
            icon: 'none',
            duration: 2000
          })
          wx.showLoading({
            title: '加载中',
            mask: true
          })
        }
      },
      complete: function () {
        wx.hideLoading();
      }
    });
  }
})

  var app = getApp(); //获取应用实例
Page({
  data: {
    type: '',
    isAuth: false,
    src: '',
    openid: '',
    unionid: '',
    height: 300,
    tips: '提示：照片验证成功！',
    isTaked: false, // 已经拍照了
    takeStatus: '0' //1成功，2失败
  },
  onLoad(option) {
    // console.log('option', option);
    if (option.type && option.type === 'employee') {
      this.setData({
        type: 'employee'
      })
    }
    const _this = this;
    const openid = app.globalData.openid;
    const unionid = app.globalData.unionid;
    // console.log('app', openid, unionid, app.globalData.guestApplyMsg);

    if (openid === '' || unionid === '') {
      app.login()
    } else {
      this.setData({
        openid,
        unionid
      })
    }
    wx.getSetting({
      success: res => {
        if (res.authSetting['scope.camera']) {
          // 用户已经授权
          _this.setData({
            isAuth: true
          })
        } else {
          // 用户还没有授权，向用户发起授权请求
          wx.authorize({
            scope: 'scope.camera',
            success() { // 用户同意授权
              _this.setData({
                isAuth: true
              })
            },
            fail() { // 用户不同意授权
              _this.openSetting().then(res => {
                _this.setData({
                  isAuth: true
                })
              })
            }
          })
        }
      },
      fail: res => {
        console.log('获取用户授权信息失败')
      }
    })
  },

  // 打开授权设置界面
  openSetting() {
    const _this = this
    let promise = new Promise((resolve, reject) => {
      wx.showModal({
        title: '授权',
        content: '请先授权获取摄像头权限',
        success(res) {
          if (res.confirm) {
            wx.openSetting({
              success(res) {
                if (res.authSetting['scope.camera']) { // 用户打开了授权开关
                  resolve(true)
                } else { // 用户没有打开授权开关， 继续打开设置页面
                  _this.openSetting().then(res => {
                    resolve(true)
                  })
                }
              },
              fail(res) {
                console.log(res)
              }
            })
          } else if (res.cancel) {
            _this.openSetting().then(res => {
              resolve(true)
            })
          }
        }
      })
    })
    return promise;
  },

  takePhoto() {
    let that = this;
    const ctx = wx.createCameraContext()
    ctx.takePhoto({
      quality: 'high',
      success: (res) => {
        console.log('res', res.tempImagePath);

        wx.showLoading({
          title: '正在上传照片...',
          mask: true
        })
        wx.uploadFile({
          //  url:app.globalData.url.login,
          url: app.globalData.host + '/upload_faceimg',
          filePath: res.tempImagePath,
          name: 'faceimg_file',
          header: {
            "Content-Type": "multipart/form-data"
          },
          success: (response) => {
            var _data = typeof response.data === 'string' ? JSON.parse(response.data) : response.data;
            wx.hideLoading();
            if (_data && _data.errcode === 0) {
              that.setData({
                src: app.globalData.host + _data.faceimg_url,
                isTaked: true,
                takeStatus: '1'
              })
            } else {
              wx.showToast({
                title: '照片上传失败',
                icon: 'none',
                duration: 2000
              })
            }

          }
        })

      }
    })
  },

  request(params) {
    const that = this;
    wx.showLoading({
      title: '数据请求中',
      mask: true
    })
    let param = params;
    if (param.errcode || param.errcode === 0) delete param.errcode;
    if (param.errmsg) delete param.errmsg;
    wx.request({
      url: app.globalData.host + '/api/v1/staff/reg',
      method: 'POST',
      data: param,
      success: res => {
        if (res.data && res.data.errcode === 0) {
          wx.showToast({
            title: '注册成功',
            icon: 'success',
            duration: 2000
          })
          wx.navigateTo({
            url: '/pages/employee/home/index/index',
          })
        } else {
          wx.showToast({
            title: '注册失败',
            icon: 'none',
            duration: 2000
          })
        }
      },
      complete: function () {
        wx.hideLoading();
      }
    })
  },
  repeat() {
    this.setData({
      src: '',
      isTaked: false
    })
  },
  ensure() {
    if (this.data.type === 'employee') { // 员工
      let userInfo = wx.getStorageSync('userInfo');
      userInfo.face_url = this.data.src;
      this.request(userInfo);

    } else { // 访客

      getApp().globalData.visit_face_url = this.data.src;
      wx.navigateTo({
        url: '/pages/guest/appointmentAffirm/appointmentAffirm',
      })
    }

  }
})
